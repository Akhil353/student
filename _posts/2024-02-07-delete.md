    <script>
        function deleteUser() {
            var uid = document.getElementById('uid').value;
            var requestBody = {
                uid: uid,
                password: password
            };
            fetch('http://127.0.0.1:8086/api/users/delete', {
                method: 'DELETE',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(requestBody),
            })
            .then(response => {
                if (response.status === 204) {
                    console.log('User deleted successfully');
                    // Do something after successful deletion, like redirecting to another page
                } else {
                    console.error('Error deleting user');
                }
            })
            .catch(error => {
                console.error('Error:', error);
            });
        }
    </script>
